\begin{frame}{Problema si context}
  \begin{itemize}
    \item In retele distribuite, mesajele trebuie propagate rapid si eficient.
    \item Exista compromis intre acoperire (delivery ratio) si overhead.
    \item Proiectul simuleaza diferite topologii si protocoale.
  \end{itemize}
\end{frame}

\begin{frame}{Protocoale: Flooding vs Gossip}
  \begin{itemize}
    \item Flooding: trimite catre toti vecinii.
    \item Gossip: trimite catre un subset aleator (parametru k).
    \item Obiectiv: compararea acoperirii cu overhead.
  \end{itemize}

\begin{frame}{Scenarii experimentale}
  \begin{itemize}
    \item Experiment 1: Comparatie protocol, fara packet loss.
    \item Experiment 2: Sweep pentru rate de pierdere.
    \item Experiment 3: Scalare dupa numar de noduri.
    \item Experiment 4: Topologii rare vs dense.
  \end{itemize}
\end{frame}

\begin{frame}{Rezultate (exemplu)}
  \begin{itemize}
    \item Flooding are delivery\_ratio mai mare, dar overhead mai ridicat.
    \item Gossip reduce overhead, dar pierde din acoperire.
    \item Packet loss afecteaza mai puternic gossip.
  \end{itemize}

  \begin{center}

  \small Overhead in topologii rare vs dense.
\end{frame}

\begin{frame}{Limitari si viitoare imbunatatiri}
  \begin{itemize}
    \item Model simplificat de trafic si topologii.
    \item Nu exista mobilitate a nodurilor.
    \item Extensii: protocoale hibride, optimizari adaptive, topologii reale.
  \end{itemize}
\end{frame}

\begin{frame}{Concluzii}
  \begin{itemize}
    \item Simulatorul ofera un cadru clar pentru compararea protocoalelor.
    \item Rezultatele arata compromisuri intre acoperire si overhead.
    \item Proiectul poate fi extins pentru scenarii mai realiste.
  \end{itemize}
